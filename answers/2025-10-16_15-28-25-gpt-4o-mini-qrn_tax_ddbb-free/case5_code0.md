from qiskit.circuit.library import HGate, XGate, Initialize

qr = QuantumRegister(3)
qc = QuantumCircuit(qr)

qc.append(XGate(), [qr[1]])

custom_gate = UnitaryGate(custom_matrix, label="Custom")

qc.append(init_gate, [qr[0]])
